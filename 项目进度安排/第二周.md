#### **2.1 文本处理流程**
```python
# backend/app/services/text_processor.py
import pdfplumber
import docx
from typing import List, Dict, Any
import re
import jieba

class TextProcessor:
    """文本处理类"""
    
    def __init__(self):
        self.chunk_size = 500  # 每个文本块500字
        self.chunk_overlap = 50  # 重叠50字
        
    def extract_text_from_pdf(self, pdf_path: str) -> str:
        """从PDF提取文本"""
        text = ""
        try:
            with pdfplumber.open(pdf_path) as pdf:
                for page in pdf.pages:
                    page_text = page.extract_text()
                    if page_text:
                        text += page_text + "\n"
        except Exception as e:
            print(f"PDF提取失败 {pdf_path}: {e}")
        return text
    
    def extract_text_from_docx(self, docx_path: str) -> str:
        """从Word文档提取文本"""
        text = ""
        try:
            doc = docx.Document(docx_path)
            for para in doc.paragraphs:
                text += para.text + "\n"
        except Exception as e:
            print(f"DOCX提取失败 {docx_path}: {e}")
        return text
    
    def clean_text(self, text: str) -> str:
        """文本清洗"""
        # 1. 去除特殊字符
        text = re.sub(r'[^\u4e00-\u9fa5a-zA-Z0-9\s，。！？；：、（）《》【】]', '', text)
        
        # 2. 合并多余空格
        text = re.sub(r'\s+', ' ', text)
        
        # 3. 去除页眉页脚（简单模式）
        lines = text.split('\n')
        cleaned_lines = []
        for line in lines:
            # 过滤页码、版权信息等
            if not re.match(r'^第[一二三四五六七八九十\d]+页', line) and \
               not re.match(r'^\d+$', line) and \
               len(line.strip()) > 10:  # 过滤太短的行
                cleaned_lines.append(line.strip())
        
        return '\n'.join(cleaned_lines)
    
    def smart_chunking(self, text: str, source_info: Dict) -> List[Dict]:
        """智能文本分块"""
        chunks = []
        
        # 1. 先按段落分割
        paragraphs = text.split('\n')
        current_chunk = ""
        current_length = 0
        chunk_index = 0
        
        for para in paragraphs:
            para = para.strip()
            if not para:
                continue
                
            para_length = len(para)
            
            # 如果当前块加上这段会超长，就保存当前块
            if current_length + para_length > self.chunk_size and current_chunk:
                chunks.append({
                    "id": f"{source_info['name']}_chunk_{chunk_index}",
                    "text": current_chunk,
                    "metadata": {
                        "source": source_info['name'],
                        "type": source_info.get('type', 'unknown'),
                        "grade": source_info.get('grade', 'unknown'),
                        "chunk_index": chunk_index,
                        "start_pos": len(current_chunk) - current_length,
                        "end_pos": len(current_chunk),
                        "word_count": len(current_chunk)
                    }
                })
                chunk_index += 1
                # 保留重叠部分
                overlap_text = current_chunk[-self.chunk_overlap:] if len(current_chunk) > self.chunk_overlap else current_chunk
                current_chunk = overlap_text + para
                current_length = len(current_chunk)
            else:
                current_chunk += para + " "
                current_length += para_length + 1
        
        # 添加最后一块
        if current_chunk:
            chunks.append({
                "id": f"{source_info['name']}_chunk_{chunk_index}",
                "text": current_chunk.strip(),
                "metadata": {
                    "source": source_info['name'],
                    "type": source_info.get('type', 'unknown'),
                    "grade": source_info.get('grade', 'unknown'),
                    "chunk_index": chunk_index,
                    "start_pos": 0,
                    "end_pos": len(current_chunk),
                    "word_count": len(current_chunk)
                }
            })
        
        return chunks
    
    def extract_keywords(self, text: str) -> List[str]:
        """提取关键词（用于元数据）"""
        # 使用结巴分词提取关键词
        words = jieba.cut(text)
        # 过滤停用词和单字
        stopwords = self.load_stopwords()
        keywords = [word for word in words if len(word) > 1 and word not in stopwords]
        
        # 返回前10个高频词
        from collections import Counter
        return [word for word, count in Counter(keywords).most_common(10)]
```

**方案A：使用百度文心Embedding API（推荐）**
```python
# backend/app/services/embedding_service.py
import requests
import json
import time
from typing import List
import logging

class BaiduEmbeddingService:
    """百度文心Embedding服务"""
    
    def __init__(self, api_key: str, secret_key: str):
        self.api_key = api_key
        self.secret_key = secret_key
        self.access_token = None
        self.token_expire_time = 0
        
    def get_access_token(self) -> str:
        """获取Access Token"""
        current_time = time.time()
        if self.access_token and current_time < self.token_expire_time:
            return self.access_token
            
        url = "https://aip.baidubce.com/oauth/2.0/token"
        params = {
            "grant_type": "client_credentials",
            "client_id": self.api_key,
            "client_secret": self.secret_key
        }
        
        try:
            response = requests.get(url, params=params)
            data = response.json()
            self.access_token = data.get("access_token")
            # Token有效期通常为30天，这里设置为25天安全期
            self.token_expire_time = current_time + 25 * 24 * 3600
            return self.access_token
        except Exception as e:
            logging.error(f"获取Access Token失败: {e}")
            raise
    
    def get_embeddings(self, texts: List[str], batch_size: int = 10) -> List[List[float]]:
        """获取文本向量"""
        access_token = self.get_access_token()
        url = f"https://aip.baidubce.com/rpc/2.0/ai_custom/v1/wenxinworkshop/embeddings/embedding-v1"
        
        all_embeddings = []
        
        # 分批处理
        for i in range(0, len(texts), batch_size):
            batch = texts[i:i+batch_size]
            
            try:
                response = requests.post(
                    f"{url}?access_token={access_token}",
                    json={"input": batch},
                    headers={"Content-Type": "application/json"}
                )
                
                if response.status_code == 200:
                    result = response.json()
                    if "data" in result:
                        batch_embeddings = [item["embedding"] for item in result["data"]]
                        all_embeddings.extend(batch_embeddings)
                    else:
                        logging.error(f"API返回格式错误: {result}")
                        # 返回零向量作为兜底
                        all_embeddings.extend([[0.0] * 384] * len(batch))
                else:
                    logging.error(f"API请求失败: {response.status_code}")
                    all_embeddings.extend([[0.0] * 384] * len(batch))
                    
                # 避免请求过快
                time.sleep(0.1)
                
            except Exception as e:
                logging.error(f"获取向量失败: {e}")
                all_embeddings.extend([[0.0] * 384] * len(batch))
        
        return all_embeddings
```

```python
from sentence_transformers import SentenceTransformer
import numpy as np

class LocalEmbeddingService:
    """本地Embedding服务"""
    
    def __init__(self, model_name: str = "paraphrase-multilingual-MiniLM-L12-v2"):
        print(f"加载本地模型: {model_name}")
        self.model = SentenceTransformer(model_name)
        print("模型加载完成")
    
    def get_embeddings(self, texts: List[str], batch_size: int = 32) -> List[List[float]]:
        """获取文本向量"""
        embeddings = self.model.encode(
            texts,
            batch_size=batch_size,
            show_progress_bar=True,
            convert_to_numpy=True
        )
        return embeddings.tolist()
```

#### **3.2 向量生成批处理脚本**
```python
# scripts/generate_embeddings.py
#!/usr/bin/env python3
"""
批量生成向量脚本
"""
import json
import os
from typing import List, Dict
import sys

# 添加项目路径
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

from backend.app.services.embedding_service import BaiduEmbeddingService
from backend.app.services.local_embedding import LocalEmbeddingService

def main():
    # 配置选择
    USE_BAIDU_API = True  # 设置为False则使用本地模型
    
    # 1. 加载处理好的文本块
    processed_dir = "./data/processed_texts"
    chunks_file = os.path.join(processed_dir, "all_chunks.json")
    
    if not os.path.exists(chunks_file):
        print("错误：未找到文本块文件")
        return
    
    with open(chunks_file, 'r', encoding='utf-8') as f:
        all_chunks = json.load(f)
    
    print(f"共加载 {len(all_chunks)} 个文本块")
    
    # 2. 初始化Embedding服务
    if USE_BAIDU_API:
        # 从环境变量读取API密钥
        api_key = os.getenv("BAIDU_API_KEY", "")
        secret_key = os.getenv("BAIDU_SECRET_KEY", "")
        
        if not api_key or not secret_key:
            print("错误：未设置百度API密钥")
            print("请在.env文件中设置 BAIDU_API_KEY 和 BAIDU_SECRET_KEY")
            return
            
        embedding_service = BaiduEmbeddingService(api_key, secret_key)
    else:
        embedding_service = LocalEmbeddingService()
    
    # 3. 分批生成向量
    batch_size = 20  # 每批处理20个文本
    all_embeddings = []
    
    for i in range(0, len(all_chunks), batch_size):
        batch = all_chunks[i:i+batch_size]
        texts = [chunk["text"] for chunk in batch]
        
        print(f"处理批次 {i//batch_size + 1}/{(len(all_chunks)+batch_size-1)//batch_size}")
        
        try:
            embeddings = embedding_service.get_embeddings(texts, batch_size=10)
            all_embeddings.extend(embeddings)
            
            # 将向量添加到文本块数据中
            for j, embedding in enumerate(embeddings):
                all_chunks[i+j]["embedding"] = embedding
                
            # 每处理100个保存一次进度
            if i % 100 == 0:
                save_progress(all_chunks[:i+batch_size])
                
        except Exception as e:
            print(f"批次处理失败: {e}")
            # 对于失败的批次，添加零向量占位
            all_embeddings.extend([[0.0] * 384] * len(texts))
    
    # 4. 保存最终结果
    output_file = "./data/embeddings/chunks_with_embeddings.json"
    os.makedirs(os.path.dirname(output_file), exist_ok=True)
    
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(all_chunks, f, ensure_ascii=False, indent=2)
    
    print(f"向量生成完成！结果保存在: {output_file}")
    print(f"总文本块数: {len(all_chunks)}")
    print(f"总向量数: {len(all_embeddings)}")

def save_progress(chunks_data: List[Dict]):
    """保存处理进度"""
    progress_file = "./data/embeddings/progress.json"
    with open(progress_file, 'w', encoding='utf-8') as f:
        json.dump(chunks_data, f, ensure_ascii=False, indent=2)

if __name__ == "__main__":
    main()
```

```python
# backend/app/services/vector_db_service.py
import chromadb
from chromadb.config import Settings
from typing import List, Dict, Any
import json
import os

class VectorDBService:
    """向量数据库服务"""
    
    def __init__(self, persist_dir: str = "./data/chroma_db"):
        self.persist_dir = persist_dir
        self.client = None
        self.collection = None
        
    def initialize(self):
        """初始化ChromaDB客户端"""
        # 创建目录
        os.makedirs(self.persist_dir, exist_ok=True)
        
        # 初始化客户端
        self.client = chromadb.PersistentClient(
            path=self.persist_dir,
            settings=Settings(
                chroma_db_impl="duckdb+parquet",
                anonymized_telemetry=False
            )
        )
        
        print(f"ChromaDB初始化完成，路径: {self.persist_dir}")
    
    def create_collection(self, collection_name: str = "history_knowledge"):
        """创建集合"""
        if not self.client:
            self.initialize()
        
        # 删除已存在的集合（可选）
        try:
            self.client.delete_collection(collection_name)
            print(f"已删除旧集合: {collection_name}")
        except:
            pass
        
        # 创建新集合
        self.collection = self.client.get_or_create_collection(
            name=collection_name,
            metadata={
                "description": "中小学历史知识向量库",
                "source": "人教版历史教材",
                "dimension": 384,  # 百度文心embedding维度
                "distance": "cosine"
            }
        )
        
        print(f"集合创建完成: {collection_name}")
        return self.collection
    
    def add_documents(self, chunks_with_embeddings: List[Dict]):
        """添加文档到向量库"""
        if not self.collection:
            self.create_collection()
        
        # 准备数据
        documents = []
        embeddings = []
        metadatas = []
        ids = []
        
        for i, chunk in enumerate(chunks_with_embeddings):
            if "embedding" not in chunk:
                print(f"警告：文本块 {i} 没有向量，跳过")
                continue
                
            documents.append(chunk["text"])
            embeddings.append(chunk["embedding"])
            metadatas.append(chunk.get("metadata", {}))
            ids.append(chunk.get("id", f"chunk_{i}"))
        
        print(f"准备添加 {len(documents)} 个文档到向量库...")
        
        # 分批添加（避免内存问题）
        batch_size = 100
        for i in range(0, len(documents), batch_size):
            batch_docs = documents[i:i+batch_size]
            batch_embeddings = embeddings[i:i+batch_size]
            batch_metadatas = metadatas[i:i+batch_size]
            batch_ids = ids[i:i+batch_size]
            
            self.collection.add(
                documents=batch_docs,
                embeddings=batch_embeddings,
                metadatas=batch_metadatas,
                ids=batch_ids
            )
            
            print(f"已添加 {min(i+batch_size, len(documents))}/{len(documents)} 个文档")
        
        print("文档添加完成！")
        
        # 验证集合大小
        count = self.collection.count()
        print(f"集合中现有 {count} 个文档")
    
    def search_similar(self, query: str, n_results: int = 5) -> List[Dict]:
        """搜索相似文档"""
        if not self.collection:
            print("错误：集合未初始化")
            return []
        
        # 这里需要先获取查询文本的向量
        # 暂时只演示搜索已有文档
        results = self.collection.query(
            query_texts=[query],
            n_results=n_results
        )
        
        # 格式化结果
        formatted_results = []
        if results["documents"]:
            for i in range(len(results["documents"][0])):
                formatted_results.append({
                    "document": results["documents"][0][i],
                    "metadata": results["metadatas"][0][i],
                    "distance": results["distances"][0][i],
                    "id": results["ids"][0][i]
                })
        
        return formatted_results
    
    def get_collection_info(self) -> Dict:
        """获取集合信息"""
        if not self.collection:
            return {}
        
        return {
            "name": self.collection.name,
            "count": self.collection.count(),
            "metadata": self.collection.metadata
        }
```

```python
def validate_vector_db():

    test_cases = [

    ]
    
    for query, expected_keyword in test_cases:
        results = vector_db.search_similar(query, n_results=3)
        assert len(results) > 0, f"查询失败: {query}"
        
        # 检查结果是否相关
        all_texts = " ".join([r["document"] for r in results])
```

```python
import PyPDF2
def extract_with_pypdf2(pdf_path):
    with open(pdf_path, 'rb') as file:
        reader = PyPDF2.PdfReader(file)
        text = ""
        for page in reader.pages:
```
